
from Cryptodome.Util.number import long_to_bytes

def reverse_binary_encoding(padd):
    m = 0
    for me in padd[::-1]:
        m = (m << 1) + (me & 1)
    return long_to_bytes(m)

with open("cipher.txt", "r") as f:
    padd = [int(x) for x in f.read().strip().split(",")]

flag = reverse_binary_encoding(padd)
print(flag)
